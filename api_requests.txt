# Start Apify Run
curl -X POST "https://api.apify.com/v2/actor-tasks/${APIFY_TASK_ID}/runs?token=${APIFY_TOKEN}" \
 -H "Content-Type: application/json" \
 -d '{"meta":{"viraltics_user_id":"user_123","connected_account_id":"acc_456"},"body":{"profileUrl":"https://www.tiktok.com/@someuser","maxItems":100,"webhookUrl":"https://<your-backend>/webhooks/apify"}}'

# Fetch Apify Dataset
curl "https://api.apify.com/v2/datasets/${DATASET_ID}/items?format=json&token=${APIFY_TOKEN}"

# Call OpenAI
curl https://api.openai.com/v1/chat/completions \
 -H "Authorization: Bearer ${OPENAI_API_KEY}" \
 -H "Content-Type: application/json" \
 -d '{"model":"gpt-4o-mini","temperature":0,"messages":[{"role":"system","content":"You are Viraltics AI assistant..."} , {"role":"user","content":"{...json input...}"}],"max_tokens":1000}'
